function hammingDistance(x: number, y: number): number {
        let result: number = 0;
        
        while (x > 0 && y > 0) {
            if ((x & 1) !== (y & 1)) {
                result++;
            }
            
            x = x >> 1;
            y = y >> 1;
        }
        
        return result;
    };
