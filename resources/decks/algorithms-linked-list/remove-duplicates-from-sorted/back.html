/**
    * Definition for singly-linked list.
    * function ListNode(val, next) {
    *     this.val = (val===undefined ? 0 : val)
    *     this.next = (next===undefined ? null : next)
    * }
    */
    /**
    * @param {ListNode} head
    * @return {ListNode}
    */
    var deleteDuplicates = function(head) {
        let h = head;
        
        while (h !== null) {
            while (h.next !== null && h.val === h.next.val) {
                h.next = h.next.next;
            }
            
            h = h.next;
        }
        
        return head;
    };
