/**
 * @param {number[]} nums
 * @return {number}
 */
var findLHS = function(nums) {
    let res = 0;
    const map = {};
    
    for (let i = 0; i < nums.length; i++) {
        map[nums[i]] = (map[nums[i]] || 0) + 1;
        if (map[nums[i] + 1]) {
            res = Math.max(res, map[nums[i]] + map[nums[i] + 1]);
        }
        if (map[nums[i] - 1]) {
            res = Math.max(res, map[nums[i]] + map[nums[i] - 1]);
        }
    }
    
    return res;
};
