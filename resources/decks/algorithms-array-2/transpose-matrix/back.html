/**
 * @param {number[][]} matrix
 * @return {number[][]}
 */
var transpose = function(matrix) {
    const result = [];
    
    for (let i = 0; i < matrix[0].length; i++) {
        for (let j = 0; j < matrix.length; j++) {
            if (!result[i]) {
                result.push([]);
            }
            
            result[i].push(matrix[j][i]);
        }
    }
    
    return result;
};
